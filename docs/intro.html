Algoritmos para el Análisis de Datos Fisiológicos

Capítulo 2 Conceptos Generales

2.1 Derivadas, derivadas parciales y sus implicaciones.

This methods tries to explain the correlation structure of a set of predictor variables using a smaller set o linear combinations of these variables called components, note that components are not variables, rather indicators of linear combinations between variables.
Given a dataset with \(m\) variables a set of \(k\) linear combinations can be used to represent it (meaning that \(k\) contains almost as much information as the \(m\) variables), also \(k<<m\).

2.2.1 TanH

TanHGener <- function(x0){
  (2/(1+exp(-2*x))-1)
}

x <- seq(-5,5,.01)

     main = "Tangente Hiperbolica TanH",
     type = "l",

2.2.4 SoftExponential

softExpGener <- function(a,x){
  if (a < 0){
    -1*((log(1-a*(x+a)))/a)
  }
  else {
    if (a == 0){
      x
    }
    else {
      ((exp(a*x)-1)/a)+a
    }
  }
}


y2 <- softExpGener(-1,x)

plot(x,y2,
     main = "SoftExponential",
     type = "l",
     col = "red",
     lwd = 4)

2.2.6 Sinc

sincGener <- function(x){
 
  xt<-x
  xt[x==0]=1
  xt[x!=0]=sin(x[x!=0])/x[x!=0]
   xt
}

y5 <- sincGener(x)
plot(x,y5,
     type = "l",
     main = "Sinc",
     col = "red",
     lwd = 4)

2.2.7 Scaled exponential linear unit (SELU)

seluGener <- function(l,a,x){
  xt<-x
  xt[x<0]=l*a*(exp(x[x<0])-1)
  xt[x>=0]=l*x[x>=0]
  xt
  #if (x < 0){
   # l*a*(exp(x)-1)
  #}
  #else {
   # l * x
  #}
}

y6 <- seluGener(1.0507, 1.67326,x)
plot(x,y6,
     type = "l",
     main = "SELU",
     col = "red",
     lwd = 4)

2.2.9 Randomized leaky rectified linear unit (RReLU)

rreluGener <- function(a,x){
  xt<-x
  xt[x<0]=a*x[x<0]
  xt
}

y9 <- rreluGener(.5,x)

plot(x,y9,
     main = "RReLU",
     type = "ln",
     col = "red",
     lwd = 4)
